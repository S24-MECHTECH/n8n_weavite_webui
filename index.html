GraphQL Templates

Klicke auf ein Template um es zu laden:

1. Lexware_rag_knowledge - Alle Objekte mit Metadaten:

{
  Get {
    Lexware_rag_knowledge(limit: 50) {
      text
      quelle
      kategorie
      titel
      _additional {
        id
        creationTimeUnix
        lastUpdateTimeUnix
      }
    }
  }
}

2. Lexware_rag_knowledge - Nach Kategorie filtern:

{
  Get {
    Lexware_rag_knowledge(where: {
      path: ["kategorie"],
      operator: Equal,
      valueText: "Buchhaltung"
    }, limit: 20) {
      text
      quelle
      kategorie
      titel
      _additional {
        id
      }
    }
  }
}

3. Lexware_import_buffer - Alle Imports:

{
  Get {
    Lexware_import_buffer(limit: 50) {
      quelle
      kategorie
      status
      datum
      _additional {
        id
        creationTimeUnix
      }
    }
  }
}

4. Aggregations - Statistiken:

{
  Aggregate {
    Lexware_rag_knowledge {
      meta {
        count
      }
      groupedBy {
        value
        count
      }
    }
  }
}
